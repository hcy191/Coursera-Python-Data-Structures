# Use the file name mbox-short.txt as the file name
fname = raw_input("Enter file name: ")
fh = open(fname)

count=0
total=0
average=0

for line in fh:
    if line.startswith("X-DSPAM-Confidence:") : 
        count = count+1
        number = float(line[line.find(":")+1:])
        total += number
        average = total/count
        
print "Average spam confidence:",average
